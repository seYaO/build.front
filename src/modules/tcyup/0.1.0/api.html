1. 用法示例

绑定调用 auto调用 
                        //获取后台上传提供的密钥，例如放在隐藏域中hideToken的值,对所有要上传的按钮,例如class为tcyup的按钮进行绑定;
                        var token = $('#hideToken');
                        $(".tcyup").tcyup({
                            token: token,//
                            title: '测试上传',//上传标题
                            pagesize: 3,//上传限制页数
                            projectid: 1234,//项目id
                            subprojectid: 1234,//项目子id
                            success: function (data) {//成功回调返回上传url数组
                                alert(data);
                            }
                        });
                        //事件触发传入初始化参数auto为true，可在事件触发是初始化。当初始化参数传入auto时，初始绑定的元素默认为1个。
                        $(".tcyup-click").on('click',function(){
                            var token = $('#hideToken');
                            $(this).tcyup({
                                token: token,
                                title: '测试上传',
                                pagesize: 3,
                                auto : true,//
                                projectid: 1234,
                                subprojectid: 1234,
                                success: function (data) {
                                    alert(data);
                                }
                            });
                        });
